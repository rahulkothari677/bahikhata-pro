    14→// ==================== USERS (Authentication) ====================
    15→model User {
    16→  id            String    @id @default(cuid())
    17→  email         String    @unique

    22→  image         String?   // profile picture URL

    24→  ownerId       String?   // if staff, links to the owner's account
    25→  createdAt     DateTime  @default(now())

    27→
    28→  // Staff permissions — JSON object controlling which modules staff can access.
    29→  permissions   Json?
    30→

    34→  renewsAt      DateTime?
    35→  trialEndsAt   DateTime?

    37→  // Relations
    38→  products      Product[]
    39→  parties       Party[]
    40→  transactions  Transaction[]
    41→  payments      Payment[]

    59→// ==================== SHOPS (Multi-shop support) ====================
    60→// Each owner can have multiple shops. Products, transactions, parties
    61→// can be tagged with a shopId. When null, shows in all shops (backward compatible).
    62→model Shop {
    63→  id          String   @id @default(cuid())
    64→  userId      String   // owner of this shop
    65→  user        User     @relation(fields: [userId], references: [id], onDelete: Cascade)
    66→  name        String   // e.g. "Sharma Kirana Store"
    67→  gstin       String?
    68→  address     String?
    69→  phone       String?
    70→  state       String?
    71→  isDefault   Boolean  @default(false)
    72→  createdAt   DateTime @default(now())
    73→  updatedAt   DateTime @updatedAt
    74→
    75→  @@index([userId])
    76→}

    78→// ==================== PRODUCTS / INVENTORY ====================
    79→model Product {
    80→  id                String   @id @default(cuid())
    81→  userId            String   // owner of this product
    82→  user              User     @relation(fields: [userId], references: [id], onDelete: Cascade)
    83→  shopId            String?  // multi-shop: null = all shops (backward compatible)
    84→  name              String
    85→  sku               String?
    86→  hsn               String?  // HSN/SAC code for GST
    87→  category          String?
    88→  unit              String   @default("pcs") // pcs, kg, ltr, m, box
    89→  purchasePrice     Float    @default(0)  // price at which shop owner buys
    90→  salePrice         Float    @default(0)  // price at which shop owner sells
    91→  mrp               Float?   // MRP printed on product
    92→  gstRate           Float    @default(0)  // 0, 5, 12, 18, 28
    93→  openingStock      Float    @default(0)
    94→  lowStockThreshold Float    @default(5)
    95→  notes             String?
    96→  createdAt         DateTime @default(now())
    97→  updatedAt         DateTime @updatedAt
    98→
    99→  transactionItems  TransactionItem[]
   100→
   101→  // Unique SKU per user (not globally)
   102→  @@unique([userId, sku])
   103→  // Performance: list products per user, search by name within user
   104→  @@index([userId, name])
   105→  @@index([userId, category])
   106→  @@index([userId, shopId])

   130→// ==================== TRANSACTIONS (Sales, Purchases, Income, Expense) ====================
   131→model Transaction {
   132→  id             String   @id @default(cuid())
   133→  userId         String   // owner of this transaction
   134→  user           User     @relation(fields: [userId], references: [id], onDelete: Cascade)
   135→  shopId         String?  // multi-shop: null = all shops (backward compatible)
   136→  invoiceNo      String?  // bill/invoice number from bill
   137→  type           String   // sale | purchase | income | expense
   138→  partyId        String?  // customer for sale, supplier for purchase
   139→  party          Party?   @relation(fields: [partyId], references: [id])
   140→  date           DateTime @default(now())
   141→  subtotal       Float    @default(0)  // before tax & discount
   142→  discountAmount Float    @default(0)
   143→  cgst           Float    @default(0)
   144→  sgst           Float    @default(0)
   145→  igst           Float    @default(0)
   146→  totalAmount    Float    @default(0)  // final payable
   147→  paidAmount     Float    @default(0)  // amount actually paid
   148→  paymentMode    String   @default("cash") // cash | upi | card | bank | credit
   149→  isInterState   Boolean  @default(false)
   150→  notes          String?
   151→  // For income/expense (no items)
   152→  category       String?  // e.g. rent, salary, electricity for expenses; commission etc for income
   153→  // Payee info for income/expense (who you paid or received from)
   154→  payeeName      String?  // name of person you paid / received from
   155→  payeePhone     String?  // mobile number of payee
   156→  createdAt      DateTime @default(now())
   157→  updatedAt      DateTime @updatedAt
   158→
   159→  items          TransactionItem[]
   160→
   161→  // Auto-computed gross profit when type=sale
   162→  grossProfit    Float    @default(0)
   163→
   164→  // Performance: list transactions per user filtered by date and/or type
   165→  // Used by: dashboard, ledger (sales/purchases), reports
   166→  @@index([userId, date])
   167→  @@index([userId, type, date])
   168→  @@index([userId, partyId])
   169→}

   171→model TransactionItem {
   172→  id             String   @id @default(cuid())
   173→  transactionId  String
   174→  transaction    Transaction @relation(fields: [transactionId], references: [id], onDelete: Cascade)
   175→  productId      String?
   176→  product        Product?  @relation(fields: [productId], references: [id])
   177→  productName    String   // snapshot of name at time of transaction
   178→  quantity       Float    @default(1)
   179→  unitPrice      Float    @default(0)  // price per unit (sale or purchase)
   180→  gstRate        Float    @default(0)
   181→  discountAmount Float    @default(0)
   182→  total          Float    @default(0)  // qty * unitPrice - discount + gst
   183→
   184→  createdAt      DateTime @default(now())
   185→}

   221→  scanLang String   @default("original")

   223→  // Controls BOTH the speech-recognition locale (e.g. mr-IN) AND the parsed
   224→  // output language. 'original' listens in hi-IN (default) and keeps the
   225→  // spoken language in the parsed result (no English translation).
   226→  voiceLang String  @default("original")

   329→// ==================== AI SCAN COMPARISONS ====================
   330→// Stores results of side-by-side AI provider tests. Each record is one
   331→// bill image scanned by 1-3 providers (Gemini, OpenAI, Groq). Admins use
   332→// the /settings/ai-comparison page to compare accuracy and pick the best.
   333→model ScanComparison {
   334→  id            String   @id @default(cuid())
   335→  userId        String
   336→  user          User     @relation(fields: [userId], references: [id], onDelete: Cascade)
   337→
   338→  // The bill image that was scanned
   339→  imageName     String?  // filename or label, if user provided one
   340→  imagePreview  String   // base64 thumbnail (max ~50KB) for display in history
   341→
   342→  // Bill type the user said it was
   343→  billType      String   @default("purchase") // sale | purchase
   344→
   345→  // Results from each provider (null if that provider wasn't tested)
   346→  geminiResult  Json?    // { success, parsed, error, durationMs, tokensUsed }
   347→  openaiResult  Json?
   348→  groqResult    Json?
   349→
   350→  // Ground truth (what the user manually entered as the correct answer)
   351→  groundTruth   Json?    // { sellerName, totalAmount, itemsCount, items: [...] }
   352→
   353→  // Computed scores per provider (0-100, null if not tested or no ground truth)
   354→  geminiScore   Float?
   355→  openaiScore   Float?
   356→  groqScore     Float?
   357→
   358→  createdAt     DateTime @default(now())
   359→
   360→  @@index([userId, createdAt])
   361→  @@index([createdAt])
   362→}

   364→// ==================== AI USAGE LOG (cost tracking) ====================
   365→// Every AI call (scan-bill, voice-parse) gets logged here with exact token
   366→// counts and calculated cost. Used by the /settings/ai-usage dashboard to
   367→// show the founder exactly how much each feature costs to run.
   368→model AiUsageLog {
   369→  id            String   @id @default(cuid())
   370→  userId        String
   371→  user          User     @relation(fields: [userId], references: [id], onDelete: Cascade)
   372→
   373→  feature       String   // 'scan-bill' | 'voice-parse' | 'scan-compare'
   374→  provider      String   // 'gemini' | 'openai' | 'groq' | 'vlm' | 'zai-sdk'
   375→  model         String   // e.g. 'gemini-2.5-flash', 'llama-4-scout-17b-16e-instruct'
   376→
   377→  // Token counts from the AI provider's response
   378→  inputTokens   Int      @default(0)
   379→  outputTokens  Int      @default(0)
   380→  totalTokens   Int      @default(0)
   381→
   382→  // Cost in INR (calculated at time of call using current pricing)
   383→  // Stored per-call so historical costs are preserved even if prices change
   384→  costInr       Float    @default(0)
   385→
   386→  // Performance + result
   387→  durationMs    Int      @default(0)
   388→  success       Boolean  @default(true)
   389→  errorMessage  String?  // null if success, error text if failed
   390→
   391→  createdAt     DateTime @default(now())
   392→
   393→  @@index([userId, createdAt])
   394→  @@index([feature, createdAt])
   395→  @@index([provider, createdAt])
   396→  @@index([createdAt])
   397→  @@index([feature, provider, createdAt])
   398→}
