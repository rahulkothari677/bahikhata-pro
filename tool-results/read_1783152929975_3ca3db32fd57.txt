   130→   130→// ==================== TRANSACTIONS (Sales, Purchases, Income, Expense) ====================
   131→   131→model Transaction {
   132→   132→  id             String   @id @default(cuid())
   133→   133→  userId         String   // owner of this transaction
   134→   134→  user           User     @relation(fields: [userId], references: [id], onDelete: Cascade)
   135→   135→  shopId         String?  // multi-shop: null = all shops (backward compatible)
   136→   136→  invoiceNo      String?  // bill/invoice number from bill
   137→   137→  type           String   // sale | purchase | income | expense
   138→   138→  partyId        String?  // customer for sale, supplier for purchase
   139→   139→  party          Party?   @relation(fields: [partyId], references: [id])
   140→   140→  date           DateTime @default(now())
   141→   141→  subtotal       Float    @default(0)  // before tax & discount
   142→   142→  discountAmount Float    @default(0)
   143→   143→  cgst           Float    @default(0)
   144→   144→  sgst           Float    @default(0)
   145→   145→  igst           Float    @default(0)
   146→   146→  totalAmount    Float    @default(0)  // final payable
   147→   147→  paidAmount     Float    @default(0)  // amount actually paid
   148→   148→  paymentMode    String   @default("cash") // cash | upi | card | bank | credit
   149→   149→  isInterState   Boolean  @default(false)
   150→   150→  notes          String?
   151→   151→  // For income/expense (no items)
   152→   152→  category       String?  // e.g. rent, salary, electricity for expenses; commission etc for income
   153→   153→  // Payee info for income/expense (who you paid or received from)
   154→   154→  payeeName      String?  // name of person you paid / received from
   155→   155→  payeePhone     String?  // mobile number of payee
   156→   156→  createdAt      DateTime @default(now())
   157→   157→  updatedAt      DateTime @updatedAt
   158→   158→
   159→   159→  items          TransactionItem[]
   160→   160→
   161→   161→  // Auto-computed gross profit when type=sale
   162→   162→  grossProfit    Float    @default(0)
   163→   163→
   164→   164→  // Performance: list transactions per user filtered by date and/or type
   165→   165→  // Used by: dashboard, ledger (sales/purchases), reports
   166→   166→  @@index([userId, date])
   167→   167→  @@index([userId, type, date])
   168→   168→  @@index([userId, partyId])
