    21→/**
    22→ * takePhotoNative — uses Capacitor Camera plugin on native (Android app)
    23→ * to open the camera directly. Falls back to <input capture> on web.
    24→ *
    25→ * Uses CameraResultType.Base64 (more reliable than DataUrl on some Android
    26→ * devices where DataUrl comes back empty after the user confirms the photo).
    27→ *
    28→ * Returns a File object or null if cancelled/failed.
    29→ */
    30→async function takePhotoNative(): Promise<File | null> {
    31→  // On native platform, use Capacitor Camera plugin
    32→  if (Capacitor.isNativePlatform()) {
    33→    try {
    34→      const { Camera, CameraResultType, CameraSource } = await import('@capacitor/camera')
    35→      const photo = await Camera.getPhoto({
    36→        quality: 85,
    37→        allowEditing: false,
    38→        resultType: CameraResultType.Base64,
    39→        source: CameraSource.Camera,
    40→        saveToGallery: false,
    41→      })
    42→      if (!photo.base64String) {
    43→        console.warn('[Camera] No base64String returned from camera')
    44→        return null
    45→      }
    46→      // Build a data URL from the base64 string (the API expects this format)
    47→      const format = photo.format || 'jpeg'
    48→      const dataUrl = `data:image/${format};base64,${photo.base64String}`
    49→      const res = await fetch(dataUrl)
    50→      const blob = await res.blob()
    51→      return new File([blob], `bill_${Date.now()}.${format}`, { type: `image/${format}` })
    52→    } catch (err: any) {
    53→      // User cancelled or permission denied — these are expected, don't log as errors
    54→      const msg = String(err?.message || err || '').toLowerCase()
    55→      if (msg.includes('cancelled') || msg.includes('user denied') || msg.includes('canceled')) {
    56→        return null
    57→      }
    58→      console.error('[Camera] Native capture failed:', err)
    59→      return null
    60→    }
    61→  }
    62→  // On web — can't open camera directly from JS, return null (caller falls back to input)
    63→  return null
    64→}

    66→/**
    67→ * pickPhotoNative — uses Capacitor Camera plugin on native (Android app)
    68→ * to open the photo gallery. Falls back to <input> on web.
    69→ */
    70→async function pickPhotoNative(): Promise<File | null> {
    71→  if (Capacitor.isNativePlatform()) {
    72→    try {
    73→      const { Camera, CameraResultType, CameraSource } = await import('@capacitor/camera')
    74→      const photo = await Camera.getPhoto({
    75→        quality: 85,
    76→        allowEditing: false,
    77→        resultType: CameraResultType.Base64,
    78→        source: CameraSource.Photos,
    79→      })
    80→      if (!photo.base64String) {
    81→        console.warn('[Camera] No base64String returned from gallery')
    82→        return null
    83→      }
    84→      const format = photo.format || 'jpeg'
    85→      const dataUrl = `data:image/${format};base64,${photo.base64String}`
    86→      const res = await fetch(dataUrl)
    87→      const blob = await res.blob()
    88→      return new File([blob], `bill_${Date.now()}.${format}`, { type: `image/${format}` })
    89→    } catch (err: any) {
    90→      const msg = String(err?.message || err || '').toLowerCase()
    91→      if (msg.includes('cancelled') || msg.includes('user denied') || msg.includes('canceled')) {
    92→        return null
    93→      }
    94→      console.error('[Camera] Native pick failed:', err)
    95→      return null
    96→    }
    97→  }
    98→  return null
    99→}

   125→    // This prevents Vercel serverless timeout on large phone photos

   127→      return new Promise((resolve, reject) => {
   128→        const reader = new FileReader()
   129→        reader.onload = (e) => {
   130→          const img = new Image()
   131→          img.onload = () => {
   132→            // Max dimensions: 1200x1600 (enough for AI to read, small enough for API)
   133→            const MAX_WIDTH = 1200
   134→            const MAX_HEIGHT = 1600
   135→            let width = img.width
   136→            let height = img.height
   137→
   138→            if (width > MAX_WIDTH || height > MAX_HEIGHT) {
   139→              const ratio = Math.min(MAX_WIDTH / width, MAX_HEIGHT / height)
   140→              width = Math.round(width * ratio)
   141→              height = Math.round(height * ratio)
   142→            }
   143→
   144→            const canvas = document.createElement('canvas')
   145→            canvas.width = width
   146→            canvas.height = height
   147→            const ctx = canvas.getContext('2d')
   148→            if (!ctx) {
   149→              reject(new Error('Canvas not supported'))
   150→              return
   151→            }
   152→            ctx.drawImage(img, 0, 0, width, height)
   153→
   154→            // Compress as JPEG with quality 0.85
   155→            const compressed = canvas.toDataURL('image/jpeg', 0.85)
   156→            resolve(compressed)
   157→          }
   158→          img.onerror = () => reject(new Error('Failed to load image'))
   159→          img.src = e.target?.result as string
   160→        }
   161→        reader.onerror = () => reject(new Error('Failed to read file'))
